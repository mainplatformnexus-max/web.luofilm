POST /payouts/quote
async function generatePayoutQuote() {
  const res = await fetch("https://function-bun-production-4110.up.railway.app/payouts/quote", {
    method: "POST",
    headers: {
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      sourceCurrency: "USD",
      destinationCurrency: "NGN",
      amount: 100,
      paymentDestination: "bank_account"
    })
  });

  const data = await res.json();
  console.log("PAYOUT QUOTE:", data);
  return data;
}
