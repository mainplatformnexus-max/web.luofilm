3. Create checkout session
async function createCheckoutSession() {
  const res = await fetch(`${API}/payments/checkout/session`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      amount: 5000,
      currency: "NGN",
      customer: {
        name: "John Doe",
        email: "john@example.com"
      },
      redirectUrl: "https://yourfrontend.com/payment-success",
      paymentMethods: ["card", "bank_transfer"],
      metadata: {
        orderId: "ORD-1001",
        userId: "USER-22"
      }
    })
  });

  const data = await res.json();
  console.log("checkout session:", data);
  return data;
}

Typical response shape:

{
  "ok": true,
  "reference": "pay-171234567890-abcd1234",
  "fincra": {
    "data": {
      "link": "https://checkout.fincra.com/...",
      "reference": "pay-171234567890-abcd1234"
    }
  }
}

How to use it on frontend:

async function startPayment() {
  const data = await createCheckoutSession();

  const checkoutLink =
    data?.fincra?.data?.link ||
    data?.fincra?.link;

  if (checkoutLink) {
    window.location.href = checkoutLink;
  } else {
    alert("No checkout link returned");
    console.log(data);
  }
}
4. Verify checkout payment

Call this after payment, usually on your success page.

async function verifyPayment(reference) {
  const res = await fetch(`${API}/payments/checkout/verify/${encodeURIComponent(reference)}`);
  const data = await res.json();
  console.log("verify payment:", data);
  return data;
}

Typical response shape:

{
  "ok": true,
  "fincra": {
    "data": {
      "status": "successful",
      "reference": "pay-171234567890-abcd1234",
      "amount": 5000
    }
  }
}

Example success-page usage:

async function checkPaymentOnPageLoad() {
  const params = new URLSearchParams(window.location.search);
  const reference = params.get("reference");

  if (!reference) return;

  const result = await verifyPayment(reference);

  const status =
    result?.fincra?.data?.status ||
    result?.fincra?.status;

  if (status === "successful") {
    document.getElementById("status").textContent = "Payment successful";
  } else {
    document.getElementById("status").textContent = "Payment not completed";
  }
}

Error handling pattern

Use this pattern for every fetch:

async function apiFetch(url, options = {}) {
  const res = await fetch(url, options);
  const data = await res.json();

  if (!res.ok || data.ok === false) {
    throw new Error(data.message || "Request failed");
  }

  return data;
}

Example:

async function getBanksSafe() {
  try {
    const data = await apiFetch(`${API}/banks`);
    console.log(data);
  } catch (error) {
    console.error("Error:", error.message);
  }
}
Important note

For security:

GET /banks is safe from frontend

POST /payments/checkout/session can be called from frontend

POST /payouts/bank and POST /payouts/mobile-money should usually be called only from your protected app flow after user auth on your own backend/database side, not openly from any public page

Because if exposed directly in a public frontend, anyone could try sending payouts.

Most common response access pattern

In many routes, your useful values will usually be inside:

data.fincra.data
